# IdCardForm
身份证表单组件

### 主要做了以下工作

 + 封装了身份证（大陆，港澳，台湾）的校验
 + 身份证表单组件的基本配置
 + 身份证校验通过返回真实用户信息

### 你可以 

 + 当作完整的身份证表单组件来使用
 

<Playground>
{
  class Demo extends React.Component {

   constructor(props) {
      super(props);
      this.state = {
        info:''
      };
  }

  onFinish(value){
    console.log('value=', value);
  };

  onSuccess(value){
    if(value){
      this.setState({
      info: JSON.stringify(value)
      })
    }
  }

  render(){
      const { info } = this.state;
      return (
        <div>
         <Form
            layout="inline"
            onFinish={(value)=>this.onFinish(value)}
          >
            <IdCardForm name='idCard' label='身份证' onSuccess={(value)=>this.onSuccess(value)}/>
            <Form.Item>
              <Button type="primary" htmlType="submit">
                提交
              </Button>
            </Form.Item>
         </Form>
          <div>{info}</div>
         </div>
      );
    }
  }
}
</Playground>
